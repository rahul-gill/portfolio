Comparing strings efficiently with hashing​

  • just comparing characters of each string: O(s1Size,s2Size)\mathcal{O}(s1Size, s2Size)O(s1Size,s2Size)
  • instead compare hash(s1)hash(s1)hash(s1) and hash(s2)hash(s2)hash(s2) in O(1)\mathcal{O}(1)O(1)(hash function return an integer)
  • if hash(s1) != hash(s2) then s1 != s2 but hash(s1) == hash(s2) doesn't always mean s1 == s2 . This is because we want to compare in O(1)\mathcal{O}(1)O(1) and for that we keep the output of hash function in fixed integer range [0,m)[0, m)[0,m)
  • polynomial rolling hash of a string s of length n is
    hash(s)=s[0]+s[1]⋅p+s[2]⋅p2+...+s[n−1]⋅pn−1mod  m\text{hash}(s) = s[0] + s[1] \cdot p + s[2] \cdot p^2 + ... + s[n-1] \cdot p^{n-1} \mod mhash(s)=s[0]+s[1]⋅p+s[2]⋅p2+...+s[n−1]⋅pn−1modm
hash(s)=∑i=0n−1s[i]⋅pimod  m,\text{hash}(s) = \sum_{i=0}^{n-1} s[i] \cdot p^i \mod m,hash(s)=i=0∑n−1​s[i]⋅pimodm,

where ppp and mmm are chosen positive numbers

  • ppp is usually taken roughly equal to number of possible characters appearing in string
  • good choice for mmm is some large prime number. for example 109+910^9+9109+9
long long rolling_hash(string const& s) {
    //we can precompute powers of p
    const int p = 31;
    const int m = 1e9 + 9;
    long long hash_value = 0;
    long long p_pow = 1;
    for (char c : s) {
        hash_value = (hash_value + (c - 'a' + 1) * p_pow) % m;
        p_pow = (p_pow * p) % m;
    }
    return hash_value;
}

Counting number of unique substring in O(n2log⁡n)\mathcal{O}(n^2\log n)O(n2logn)​

int count_unique_substrings(string const& s) {
    int n = s.size();

    const int p = 31;
    const int m = 1e9 + 9;

    //precompute powers of p
    vector<long long> p_pow(n);
    p_pow[0] = 1;
    for (int i = 1; i < n; i++)
        p_pow[i] = (p_pow[i-1] * p) % m;

    //hashes of prefixes
    vector<long long> h(n + 1, 0);
    for (int i = 0; i < n; i++)
        h[i+1] = (h[i] + (s[i] - 'a' + 1) * p_pow[i]) % m;

    int cnt = 0;
    for (int l = 1; l <= n; l++) {
        set<long long> hs;
        for (int i = 0; i <= n - l; i++) {
            long long cur_h = (h[i + l] + m - h[i]) % m;
            //need to understand this part
            cur_h = (cur_h * p_pow[n-i-1]) % m;
            hs.insert(cur_h);
        }
        cnt += hs.size();
    }
    return cnt;
}

Rabin Karp Algorithm​

vector<int> rabin_karp(string const& s, string const& t) {
    const int p = 31; 
    const int m = 1e9 + 9;
    int S = s.size(), T = t.size();
    
    //precompute powers of p
    vector<long long> p_pow(max(S, T)); 
    p_pow[0] = 1; 
    for (int i = 1; i < (int)p_pow.size(); i++) 
        p_pow[i] = (p_pow[i-1] * p) % m;

    //calculating hashes of prefixes of t
    vector<long long> h(T + 1, 0); 
    for (int i = 0; i < T; i++)
        h[i+1] = (h[i] + (t[i] - 'a' + 1) * p_pow[i]) % m; 
    
    //calculating hash of pattern
    long long h_s = 0; 
    for (int i = 0; i < S; i++) 
        h_s = (h_s + (s[i] - 'a' + 1) * p_pow[i]) % m; 

    vector<int> occurences;
    for (int i = 0; i + S - 1 < T; i++) { 
        long long cur_h = (h[i+S] + m - h[i]) % m; 
        if (cur_h == h_s * p_pow[i] % m)
            occurences.push_back(i);
    }
    return occurences;
}

    long long t=0,p=0,h=1;

        }
    }
    if(v.empty())

vector<int> prefix_function(const string& s) {
    int n = (int)s.size();
    vector<int> pi(n);
    for (int i = 1; i < n; i++) {
        int j = pi[i-1];
        while (j > 0 && s[i] != s[j])
            j = pi[j-1];
        if (s[i] == s[j])
            j++;
        pi[i] = j;
    }
    return pi;
}

Knuth Morris Pratt Algorithm​

  • problem: Given a text ttt and a string sss, we want to find and display the positions of all occurrences of the string sss in the text ttt . length of sss = nnn, length of ttt = mmm
  • solution with prefix function: we generate the string s+#+ts + \# + ts+#+t and compute the prefix function for it. then if at position iii of it we have π[i]=n\pi[i] = nπ[i]=n then the string sss appears in ttt at position i−(n+1)−n+1=i−2ni - (n + 1) - n + 1 = i - 2ni−(n+1)−n+1=i−2n
  • since π[i]\pi[i]π[i] doesn't exceeds nnn so the KMP Algorithm solves it in O(n+m)\mathcal{O}(n+m)O(n+m) in time and O(n)\mathcal{O}(n)O(n) in memory
vector<int> kmp(const string& t,const string& s){
    vector<int> ret;
    vector<int> pi = prefix_function(s);
    int j = 0;
    int n = (int)s.size();
    
    for(int i=0; i< t.size(); i++){
        while (j > 0 && s[j] != t[i] )
            j = pi[j-1];
        if(s[j] == t[i])
            j++;
            
        if(j == n)
            ret.push_back(i - (n-1));
    }
    return ret;
}

Boyer Moore Algorithm​

  • Bad char rule: Upon mismatch, skip alignments until that mismatch becomes a match or we move past the mistmatch
  • Good suffix rule:Upon mismatch, skip alignments until that mismatch becomes a match or we move past the mistmatch
#define NO_OF_CHARS 256

// O(nm) worst O(n/m) best O(n) avg
void bad_char_util(const string& str, int badchar[NO_OF_CHARS]){
    int i;
    for (i = 0; i < NO_OF_CHARS; i++)
        badchar[i] = str.size();
    for (i = 0; i < str.size() - 1; i++)
        badchar[(int) str[i]] = i;
}

vector<int> boyer_moore(const string& t,const string& s){
    vector<int> ans;
    int badchar[NO_OF_CHARS];
    bad_char_util(s, badchar);

    for(int i=0; i <= t.size()-s.size();){
        int j = s.size() - 1;
        while(j>=0 && t[i+j] == s[j])
            --j;
        if(j<0){
            ans.push_back(i++);
            i += s.size()-badchar[t[i + s.size()]];
        }
        else
            i += max(1, j - badchar[t[i + j]]);
    }
    return ans;
}
