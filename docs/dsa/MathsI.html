long long binexp(long long base, long long index, long long m){
    base %= m;
    long long res = 1;
    while(index > 0){
        if(index & 1) res = (res * base) % m;
        base = (base * base) % m;
        index >>= 1;
    }
    return res;
}

Stein's GCD algorithm(removes slower modulo operations)​

  • if both are even remove both's 2i
  • if one in even and other odd, remove even's 2i
  • if both are odd then bigger = bigger - smaller
long long gcd(long long a, long long b) { 
    if (!a || !b) 
        return a | b; 
    unsigned long long shift = __builtin_ctzll(a | b); 
    a >>= __builtin_ctzll(a); 
    do { 
        b >>= __builtin_ctzll(b); 
        if (a > b) 
            swap(a, b); 
        b -= a; 
    } while (b); 
    return a << shift; 
}

int extended_gcd(int a, int b, int& x, int& y) { 
    x = 1, y = 0; 
    int x1 = 0, y1 = 1; 
    while (b) { 
        int q = a / b; 
        tie(x, x1) = make_tuple (x1, x - q * x1); 
        tie(y, y1) = make_tuple (y1, y - q * y1); 
        tie(a, b) = make_tuple (b, a - q * b); 
    } 
    return a; 
}

bool find_any_solution (int a, int b, int c, int &x0, int &y0) {
    int g = extended_gcd(abs(a), abs(b), x0, y0);

int find_all_solutions(int a, int b, int c, int minx, int maxx, int miny, int maxy) {

    if (x < minx)

    if (x > maxx)

    if (x > maxx)
        shift_solution(x, y, a, b, -sign_b);
    int rx1 = x;

    shift_solution(x, y, a, b, -(miny - y) / a);
    if (y < miny)
        shift_solution(x, y, a, b, -sign_a);
    if (y > maxy)
        return 0;
    int lx2 = x;

    if (y > maxy)

    int rx2 = x;

    return (rx - lx) / abs(b) + 1;

pair<int, int> fib (int n) {
    if (n == 0)
        return {0, 1};

    auto p = fib(n >> 1);
    int c = p.first * (2 * p.second - p.first);
    int d = p.first * p.first + p.second * p.second;
    if (n & 1)
        return {d, c + d};
    else
        return {c, d};
}
